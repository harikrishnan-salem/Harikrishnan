Rock Paper Scissors

import random  

choices = {"r": "rock", "p": "paper", "s": "scissors"}

total_pw, total_cw = 0, 0

while True:
    while True:
        try:
            R = int(input("How many rounds do you want to win: "))
            if R > 0:
                break
            else:
                print("Please enter a positive number!")
        except ValueError:
            print("Wrong input! Please enter a number.")

    pw, cw = 0, 0
    round_num = 1
    print(f"\nüéØ First to {R} wins!")

    while pw < R and cw < R:
        print(f"\n--- Round {round_num} ---")
        player = input("Choose (r)ock, (p)aper, (s)cissors: ").lower()
        if player not in choices:
            print("‚ùå Invalid choice, try again!")
            continue

        player_choice = choices[player]
        computer_choice = random.choice(list(choices.values()))
        print(f"üßë Player choice: {player_choice}")  
        print(f"ü§ñ Computer choice: {computer_choice}")  

        if player_choice == computer_choice:
            print("‚öñÔ∏è It's a Tie!")
        elif (player_choice == "rock" and computer_choice == "scissors") or \
             (player_choice == "scissors" and computer_choice == "paper") or \
             (player_choice == "paper" and computer_choice == "rock"):
            pw += 1
            print("‚úÖ Player wins this round!")
        else:
            cw += 1
            print("‚ùå Computer wins this round!")

        print(f"üìä Score ---> Player: {pw}, Computer: {cw}")  
        round_num += 1

    print("\n" + "="*30)
    if pw > cw:
        print("üéâ Player wins the game!")
        total_pw += 1
    else:
        print("ü§ñ Computer wins the game!")
        total_cw += 1
    print("="*30)
    print(f"üèÜ Match Score (All games) ---> Player: {total_pw}, Computer: {total_cw}")

    while True:
        again = input("\nPlay again? (yes/no): ").strip().lower()
        if again in ["yes", "y"]:
            break
        elif again in ["no", "n"]:
            print("üëã Thanks for playing!")
            exit()
        else:
            print("‚ùå Invalid input. Please type yes or no.")

Number Guess

import random

print("üéØ Welcome to the Number Guessing Game!")
print("Think of a number between 1 and 100, and I will try to guess it.")

low = 1
high = 100
attempts = 0

while True:
    guess = random.randint(low, high)
    attempts += 1
    print(f"My guess is: {guess}")

    feedback = input("Is it correct (c), higher (h), or lower (l)? ").lower()
    if feedback == "c":
        print(f"üéâ Yay! I guessed your number in {attempts} tries!")
        break
    elif feedback == "h":
        low = guess + 1
    elif feedback == "l":
        high = guess - 1
    else:
        print("‚ö† Please enter only 'c', 'h', or 'l'.")
